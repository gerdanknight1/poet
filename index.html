def clamp(x: float, lo: float, hi: float) -> float:

    return max(lo, min(hi, x))



def fmt1(x: float) -> str:

    return f"{x:.1f}"



def now_str() -> str:

    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")



def exp_needed_for_next(level: int) -> int:

    return 20 + (level - 1) * 8



def grade_for_quality(q: int) -> str:

    if q < 20:

        return "초고(거칠음)"

    if q < 35:

        return "습작(보통)"

    if q < 50:

        return "준작(괜찮음)"

    if q < 70:

        return "수작(좋음)"

    if q < 90:

        return "명작(매우 좋음)"

    return "걸작(레전드)"



def base_fame_for_quality(q: int) -> float:

    # 45 -> 평균 4.5

    base = q / 10.0

    jitter = random.uniform(-0.10, 0.10) * base

    return round(clamp(base + jitter, 0.0, 1e18), 1)

        "대형서점": "대형출판사",

def normalize_type(raw: str) -> str | None:

    if not raw:

        return None

    s = raw.strip().replace(" ", "")

@dataclass

class Work:

    id: int

    title: str

    wtype: str

    quality: int

    grade: str

    created_at: str

    published: bool = False

    platform: str = ""

    fame_gained: float = 0.0

    published_at: str = ""



@dataclass

class State:

    level: int = 1

    exp: int = 0

    fame: float = 0.0

    blueprints: dict | None = None

    works: list[dict] | None = None

    awards_owned: dict | None = None  # cost(str) -> count

    next_work_id: int = 1


    def __post_init__(self):

        if self.blueprints is None:

            self.blueprints = {t: 0 for t in BLUEPRINT_TYPES}

        else:

            for t in BLUEPRINT_TYPES:

                self.blueprints.setdefault(t, 0)


        if self.works is None:

            self.works = []


        if self.awards_owned is None:

            self.awards_owned = {str(cost): 0 for cost, _name, _fx in AWARDS}

        else:

            for cost, _name, _fx in AWARDS:

                self.awards_owned.setdefault(str(cost), 0)



class Game:

    def __init__(self):

        self.s = State()

        self.running = True


        # 아카이브 옵션(저장 X)

        self.archive_sort_mode = "new"   # new|old|q|fame

        self.archive_page_size = 30

    def prompt(self) -> str:

        except (EOFError, KeyboardInterrupt):

            print()

            return "quit"


    # ---------- basics ----------

    def help(self):

        self.say("train                          : 수련(경험치)")

        self.say("inspire                        : 영감 얻기(제작서)")

        self.say("blueprints                     : 제작서 현황")

        self.say("write <타입> <제목>             : 작품 작성")

        self.say("works                          : 최근 작품 30개")

        self.say("archive                        : 전체 작품 아카이브")

        self.say("publish <id> <플랫폼>           : 작품 발표(조건 있음)")

        self.say("platforms                      : 플랫폼/배율/조건 보기")

        self.say("awards                         : 문학상 목록/가격 보기")

        self.say("buy <50|500|5000|50000>         : 문학상 구매(패시브 누적)")

        self.say("effects                        : 현재 패시브 총합 보기")

        self.say("save / load                    : 저장 / 불러오기")

        self.say("quit                           : 종료")

        self.say("")

        self.say("타입: " + ", ".join(BLUEPRINT_TYPES))

        self.say("플랫폼: 인스타 / 신진 문예지 / 중앙 문예지 / 대형 서점 / 글로벌 출판사")

        self.say("아카이브 도움: archive")


    def status(self):

        need = exp_needed_for_next(self.s.level)

        self.say("=== 상태 ===")

        self.say(f"레벨: {self.s.level}")

        self.say(f"경험치: {self.s.exp}/{need} (다음 레벨까지 {max(0, need - self.s.exp)})")

        self.say(f"명성: {fmt1(self.s.fame)}")

        self.say(f"작품 수: {len(self.s.works)}")

        self.say("")

        self.say("제작서:")

        for t in BLUEPRINT_TYPES:

            self.say(f" - {t}: {self.s.blueprints.get(t, 0)}")

        self.say("")

        self.say("문학상 보유:")

        for cost, name, _fx in AWARDS:

            cnt = int(self.s.awards_owned.get(str(cost), 0))

            self.say(f" - {name} ({cost}): {cnt}")


    def show_blueprints(self):

        self.say("=== 제작서 보유 ===")

        for t in BLUEPRINT_TYPES:

            self.say(f"{t}: {self.s.blueprints.get(t, 0)}")


    def show_platforms(self):

        self.say("=== 발표 플랫폼 (배율 / 조건) ===")

        for _key, info in PLATFORMS.items():

            label = info["label"]

            mult = info["mult"]

            min_q = info["min_q"]

            cond = "제한 없음" if min_q <= 0 else f"퀄리티 {min_q} 이상"

            self.say(f"- {label}: x{mult} | 조건: {cond}")

        self.say("")

        self.say("예: publish 3 대형서점  (또는 publish 3 대형출판사 / 대형)")


    def show_awards(self):

        self.say("=== 문학상 상점(반값 적용) ===")

        for cost, name, _fx in AWARDS:

            self.say(f"- {name} : {cost} 명성")

        self.say("buy 50 / buy 500 / buy 5000 / buy 50000")

        self.say("패시브 확인: effects")


    # ---------- passives ----------

    def compute_passives(self) -> Dict[str, float]:

        exp_mult = 1.0

        fame_mult = 1.0

        exp_flat = 0.0

        quality_flat = 0.0

        inspire_plus1_prob = 0.0


        for cost, _name, fx in AWARDS:

            cnt = int(self.s.awards_owned.get(str(cost), 0))

            if cnt <= 0:

                continue

            exp_mult *= float(fx["exp_mult"]) ** cnt

            fame_mult *= float(fx["fame_mult"]) ** cnt

            exp_flat += float(fx["exp_flat"]) * cnt

            quality_flat += float(fx["quality_flat"]) * cnt

            inspire_plus1_prob += float(fx["inspire_plus1_prob"]) * cnt


        inspire_plus1_prob = clamp(inspire_plus1_prob, 0.0, 0.95)

        return {

            "exp_mult": exp_mult,

            "fame_mult": fame_mult,

            "exp_flat": exp_flat,

            "quality_flat": quality_flat,

            "inspire_plus1_prob": inspire_plus1_prob,

        }


    def show_effects(self):

        fx = self.compute_passives()

        self.say("=== 현재 패시브 총합 ===")

        self.say(f"수련 경험치 배율: x{fx['exp_mult']:.3f}")

        self.say(f"수련 경험치 추가: +{int(fx['exp_flat'])}")

        self.say(f"작품 퀄리티 추가: +{int(fx['quality_flat'])}")

        self.say(f"발표 명성 배율: x{fx['fame_mult']:.3f}")

        self.say(f"영감 추가 제작서(+1) 확률: {fx['inspire_plus1_prob']*100:.1f}%")


    # ---------- save/load ----------

    def save(self):

        with open(SAVE_PATH, "w", encoding="utf-8") as f:

            json.dump(asdict(self.s), f, ensure_ascii=False, indent=2)

        self.say(f"저장 완료: {SAVE_PATH}")


    def load(self):

        if not os.path.exists(SAVE_PATH):

            self.say("세이브 파일이 없어. 먼저 save 해줘.")

            return

        with open(SAVE_PATH, "r", encoding="utf-8") as f:

            data = json.load(f)

        self.s = State(**data)

        self.say("로드 완료. 이어서 진행한다.")


    # ---------- actions ----------

    def maybe_level_up(self):

        while True:

            need = exp_needed_for_next(self.s.level)

            if self.s.exp < need:

                break

            self.s.exp -= need

            self.s.level += 1

            self.say(f"레벨업! -> Lv.{self.s.level}")


    def train(self):

        fx = self.compute_passives()

        base_gain = random.randint(8, 14)

        gain = int(round(base_gain * fx["exp_mult"] + fx["exp_flat"]))

        if gain < 1:

            gain = 1

        self.s.exp += gain

        self.say(f"수련했다. 경험치 +{gain} (기본 {base_gain})")

        self.maybe_level_up()


    def inspire(self):

        fx = self.compute_passives()

        count = 1 + (1 if random.random() < 0.25 else 0)

        if random.random() < fx["inspire_plus1_prob"]:

            count += 1


        gained_types = []

        for _ in range(count):

            t = random.choice(BLUEPRINT_TYPES)

            self.s.blueprints[t] = self.s.blueprints.get(t, 0) + 1

            gained_types.append(t)


        vibes = [

            "지하철 유리창에 비친 얼굴이 문장이 됐다.",

            "편의점 형광등 아래서 한 줄이 떠올랐다.",

            "새벽 알림 소리 사이로 리듬이 지나갔다.",

            "뉴스 헤드라인이 은근히 비유가 됐다.",

            "낡은 메모장에서 제목이 먼저 튀어나왔다.",

        ]

        self.say(f"영감을 얻었다. 제작서 +{count} ({', '.join(gained_types)})")

        self.say(f"  - {random.choice(vibes)}")


    def write(self, wtype_raw: str, title: str):

        wtype = normalize_type(wtype_raw)

        if wtype is None:

            self.say("타입을 못 알아먹었어.")

            self.say("가능 타입: " + ", ".join(BLUEPRINT_TYPES))

            self.say("예: write 서정 비 오는 날의 무릎")

            return


        title = (title or "").strip()

        if not title:

            self.say("제목이 필요해. 예: write 리얼리즘 퇴근길의 먼지")

            return


        if self.s.blueprints.get(wtype, 0) <= 0:

            self.say(f"'{wtype}' 제작서가 없어. inspire로 먼저 얻어.")

            return


        self.s.blueprints[wtype] -= 1

        fx = self.compute_passives()

        q = sum(random.randint(6, 10) for _ in range(self.s.level)) + int(fx["quality_flat"])

        g = grade_for_quality(q)


        w = Work(

            id=self.s.next_work_id,

            title=title,

            wtype=wtype,

            quality=q,

            grade=g,

            created_at=now_str(),

        )

        self.s.next_work_id += 1

        self.s.works.append(asdict(w))


        self.say("작품을 썼다.")

        self.say(f" - ID: {w.id}")

        self.say(f" - 타입: {w.wtype}")

        self.say(f" - 제목: {w.title}")

        self.say(f" - 퀄리티: {w.quality} ({w.grade})")


    def _format_work_line(self, wd: dict) -> str:

        pub = "발표됨" if wd.get("published") else "미발표"

        extra = ""

        if wd.get("published"):

            extra = f" | {wd.get('platform','')} | 명성 +{fmt1(wd.get('fame_gained', 0.0))}"

        return (

            f"[{wd['id']}] ({wd.get('wtype','?')}) {wd['title']} | "

            f"Q{wd['quality']} {wd['grade']} | {pub}{extra}"

        )


    def list_works_recent(self):

        if not self.s.works:

            self.say("아직 작품이 없어. inspire → write로 만들어봐.")

            return

        self.say("=== 작품 목록(최근 30개) ===")

        for wd in self.s.works[-30:]:

            self.say(self._format_work_line(wd))

        if len(self.s.works) > 30:

            self.say(f"(최근 30개만 표시됨. 총 {len(self.s.works)}개)  → 전체는 archive")


    def publish(self, wid: int, platform_raw: str):

        pkey = normalize_platform(platform_raw)

        if pkey is None:

            self.say("플랫폼을 못 알아먹었어. platforms로 목록을 확인해줘.")

            return

        pinfo = PLATFORMS[pkey]


        for wd in self.s.works:

            if wd["id"] != wid:

                continue


            if wd.get("published"):

                self.say("이미 발표한 작품이야.")

                return


            q = int(wd["quality"])

            min_q = int(pinfo["min_q"])

            if q < min_q:

                self.say("발표 실패: 플랫폼 조건을 못 맞췄어.")

                self.say(f" - 플랫폼: {pinfo['label']} (필요 퀄리티 {min_q}+)")

                self.say(f" - 현재 퀄리티: {q}")

                return


            fx = self.compute_passives()

            base = base_fame_for_quality(q)

            gain = round(base * float(pinfo["mult"]) * float(fx["fame_mult"]), 1)


            wd["published"] = True

            wd["platform"] = pinfo["label"]

            wd["fame_gained"] = gain

            wd["published_at"] = now_str()


            self.s.fame = round(self.s.fame + gain, 1)


            self.say("작품을 발표했다.")

            self.say(f" - 제목: {wd['title']}")

            self.say(f" - 타입: {wd.get('wtype','?')}")

            self.say(f" - 퀄리티: {q} ({wd['grade']})")

            self.say(f" - 플랫폼: {pinfo['label']} (x{pinfo['mult']})")

            self.say(f" - 패시브 명성배율: x{fx['fame_mult']:.3f}")

            self.say(f" - 획득 명성: +{fmt1(gain)}  (누적 명성: {fmt1(self.s.fame)})")

            return


        self.say("그 ID의 작품을 찾지 못했어.")


    def buy(self, cost: int):

        valid = {c for c, _name, _fx in AWARDS}

        if cost not in valid:

            self.say("가격은 50, 500, 5000, 50000 중 하나만 가능해.")

            return

        if self.s.fame < cost:

            self.say(f"명성이 부족해. 필요: {cost}, 현재: {fmt1(self.s.fame)}")

            return


        self.s.fame = round(self.s.fame - cost, 1)

        self.s.awards_owned[str(cost)] = int(self.s.awards_owned.get(str(cost), 0)) + 1


        name = next(n for c, n, _fx in AWARDS if c == cost)

        self.say(f"{name}을(를) 구매했다. (-{cost} 명성)")

        self.say(f"남은 명성: {fmt1(self.s.fame)}")


    # ---------- ARCHIVE ----------

    def archive_help(self):

        self.say("=== 아카이브(전체 작품 열람) ===")

        self.say("archive page <n>            : 30개씩 페이지 보기 (1=가장 최근)")

        self.say("archive show <id>           : 특정 작품 상세 보기")

        self.say("archive find <키워드>        : 제목 검색(부분 일치)")

        self.say("archive filter type <타입>   : 타입별 보기")

        self.say("archive filter pub <y|n>     : 발표(y)/미발표(n)만 보기")

        self.say("archive sort <new|old|q|fame>: 정렬 기준 설정(아카이브 출력에 적용)")


    def _sorted_items(self, items: List[dict]) -> List[dict]:

        m = self.archive_sort_mode

        if m == "new":

            return sorted(items, key=lambda w: int(w.get("id", 0)), reverse=True)

        if m == "old":

            return sorted(items, key=lambda w: int(w.get("id", 0)))

        if m == "q":

            return sorted(items, key=lambda w: (int(w.get("quality", 0)), int(w.get("id", 0))), reverse=True)

        if m == "fame":

            return sorted(items, key=lambda w: (float(w.get("fame_gained", 0.0)), int(w.get("id", 0))), reverse=True)

        return items


    def archive_page(self, page: int):

        if not self.s.works:

            self.say("아직 작품이 없어.")

            return

        if page < 1:

            self.say("페이지는 1 이상이야.")

            return


        items = self._sorted_items(list(self.s.works))

        total = len(items)

        per = self.archive_page_size

        max_page = (total + per - 1) // per


        if page > max_page:

            self.say(f"페이지 범위 초과. 최대 페이지: {max_page}")

            return


        start = (page - 1) * per

        end = min(start + per, total)

        self.say(f"=== 아카이브 페이지 {page}/{max_page} (정렬: {self.archive_sort_mode}) ===")

        for wd in items[start:end]:

            self.say(self._format_work_line(wd))

        self.say(f"(표시: {start+1}-{end} / 전체 {total})")


    def archive_show(self, wid: int):

        for wd in self.s.works:

            if int(wd.get("id", -1)) == wid:

                self.say("=== 작품 상세 ===")

                self.say(f"ID: {wd['id']}")

                self.say(f"제목: {wd['title']}")

                self.say(f"타입: {wd.get('wtype','?')}")

                self.say(f"생성일: {wd.get('created_at','')}")

                self.say(f"퀄리티: {wd.get('quality','?')} ({wd.get('grade','')})")

                self.say(f"상태: {'발표됨' if wd.get('published') else '미발표'}")

                if wd.get("published"):

                    self.say(f"플랫폼: {wd.get('platform','')}")

                    self.say(f"발표일: {wd.get('published_at','')}")

                    self.say(f"획득 명성: +{fmt1(float(wd.get('fame_gained', 0.0)))}")

                return

        self.say("그 ID의 작품을 찾지 못했어.")


    def archive_find(self, keyword: str):

        kw = (keyword or "").strip()

        if not kw:

            self.say("예: archive find 고양이")

            return

        hits = [w for w in self.s.works if kw.lower() in str(w.get("title", "")).lower()]

        hits = self._sorted_items(hits)

        self.say(f"=== 검색 결과: '{kw}' ({len(hits)}개) (정렬: {self.archive_sort_mode}) ===")

        for wd in hits[:100]:

            self.say(self._format_work_line(wd))

        if len(hits) > 100:

            self.say("(검색 결과가 많아서 100개까지만 표시)")


    def archive_filter_type(self, t_raw: str):

        t = normalize_type(t_raw)

        if t is None:

            self.say("타입을 못 알아먹었어.")

            self.say("가능 타입: " + ", ".join(BLUEPRINT_TYPES))

            return

        hits = [w for w in self.s.works if w.get("wtype") == t]

        hits = self._sorted_items(hits)

        self.say(f"=== 타입 필터: {t} ({len(hits)}개) (정렬: {self.archive_sort_mode}) ===")

        for wd in hits[:200]:

            self.say(self._format_work_line(wd))

        if len(hits) > 200:

            self.say("(결과가 많아서 200개까지만 표시)")


    def archive_filter_pub(self, yn: str):

        s = (yn or "").strip().lower()

        if s not in ("y", "n"):

            self.say("예: archive filter pub y  (또는 n)")

            return

        want = (s == "y")

        hits = [w for w in self.s.works if bool(w.get("published")) == want]

        hits = self._sorted_items(hits)

        label = "발표" if want else "미발표"

        self.say(f"=== 상태 필터: {label} ({len(hits)}개) (정렬: {self.archive_sort_mode}) ===")

        for wd in hits[:200]:

            self.say(self._format_work_line(wd))

        if len(hits) > 200:

            self.say("(결과가 많아서 200개까지만 표시)")


    def archive_sort(self, mode: str):

        m = (mode or "").strip().lower()

        if m not in ("new", "old", "q", "fame"):

            self.say("예: archive sort new | old | q | fame")

            return

        self.archive_sort_mode = m

        self.say(f"아카이브 정렬을 '{m}'로 설정했어. (archive page 1)")


    def handle_archive(self, args: List[str]):

        if not args:

            self.archive_help()

            return


        sub = args[0].lower()

        if sub == "page":

            if len(args) < 2:

                self.say("예: archive page 1")

                return

            try:

                p = int(args[1])

            except ValueError:

                self.say("페이지는 숫자야.")

                return

            self.archive_page(p)

            return


        if sub == "show":

            if len(args) < 2:

                self.say("예: archive show 12")

                return

            try:

                wid = int(args[1])

            except ValueError:

                self.say("ID는 숫자야.")

                return

            self.archive_show(wid)

            return


        if sub == "find":

            self.archive_find(" ".join(args[1:]))

            return


        if sub == "filter":

            if len(args) < 3:

                self.say("예: archive filter type 서정")

                self.say("    archive filter pub y")

                return

            kind = args[1].lower()

            rest = " ".join(args[2:])

            if kind == "type":

                self.archive_filter_type(rest)

                return

            if kind == "pub":

                self.archive_filter_pub(rest)

                return

            self.say("filter 종류는 type 또는 pub만 지원해.")

            return


        if sub == "sort":

            if len(args) < 2:

                self.say("예: archive sort q")

                return

            self.archive_sort(args[1])

            return


        self.say("아카이브 명령을 못 알아먹었어. archive 로 도움말을 봐줘.")

    def run(self):

        self.say("=== 현대에서 시인 키우기 ===")

        self.say("help / platforms / awards / effects / archive")

        while self.running:

            cmdline = self.prompt()

            if not cmdline:

                continue

            parts = cmdline.split()

            cmd = parts[0].lower()

            args = parts[1:]


            if cmd in ("quit", "exit", "q"):

                self.running = False

                self.say("종료.")

            elif cmd == "help":

                self.help()

            elif cmd == "status":

                self.status()

            elif cmd == "effects":

                self.show_effects()

            elif cmd == "train":

                self.train()

            elif cmd == "inspire":

                self.inspire()

            elif cmd == "blueprints":

                self.show_blueprints()

            elif cmd == "write":

                if len(args) < 2:

                    self.say("예: write 서정 비 오는 날의 무릎")

                    continue

                wtype_raw = args[0]

                title = cmdline.split(None, 2)[2]

                self.write(wtype_raw, title)

            elif cmd == "works":

                self.list_works_recent()

            elif cmd == "archive":

                self.handle_archive(args)

            elif cmd == "platforms":

                self.show_platforms()

            elif cmd == "publish":

                if len(args) < 2:

                    self.say("예: publish 3 대형서점")

                    continue

                try:

                    wid = int(args[0])

                except ValueError:

                    self.say("작품 ID는 숫자야.")

                    continue

                platform_raw = "".join(args[1:])

                self.publish(wid, platform_raw)

            elif cmd == "awards":

                self.show_awards()

            elif cmd == "buy":

                if not args:

                    self.say("예: buy 5000")

                    continue

                try:

                    cost = int(args[0])

                except ValueError:

                    self.say("예: buy 50 / buy 500 / buy 5000 / buy 50000")

                    continue

                self.buy(cost)

            elif cmd == "save":

                self.save()

            elif cmd == "load":

                self.load()

            else:

                self.say("모르는 명령어야. help 를 쳐봐.")
